int maxSubArray(int* nums, int numsSize)
{
    int max = nums[0];
    int curr = nums[0];
    for(int i = 1; i < numsSize; i++)
    {        
        curr += nums[i];                    /* merge next */
        if (curr < 0 || curr < nums[i])     /* merge is NOT good idea */
            curr = nums[i];
        if (max < curr)                     /* reserve history max */
            max = curr;
    }
    return max;        
}
