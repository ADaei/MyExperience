int maxSubArray(int* nums, int numsSize)
{
    int max = nums[0];
    int cur = nums[0];
    for(int i = 1; i < numsSize; i++)
    {        
        cur += nums[i];                     /* merge next */
        if (cur < 0 || cur < nums[i])       /* merge is NOT good idea */
            cur = nums[i];
        if (max < cur)                      /* reserve history max */
            max = cur;
    }
    return max;        
}
